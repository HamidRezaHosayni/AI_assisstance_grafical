# 1. ایجاد محیط مجازی (اگر نداری)
python -m venv myai_env

# 2. فعال کردن محیط مجازی
# ویندوز:
myai_env\Scripts\activate
# مک/لینوکس:
source myai_env/bin/activate

# 3. نصب کتابخانه‌های اصلی
pip install PyQt6 PyQt6-WebEngine PyQt6-Svg

# 4. نصب LangChain و ابزارهای مرتبط
pip install langchain langchain-community langchain-core

# 5. نصب FAISS برای جستجوی برداری
pip install faiss-cpu  # یا faiss-gpu اگر GPU داری

# 6. نصب Ollama SDK برای اتصال به مدل‌های لوکال
pip install ollama

# 7. نصب موتور STT/TTS (ماژول صوتی)
pip install speechrecognition pyttsx3

# 8. نصب کتابخانه‌های کمکی
pip install python-dotenv pillow numpy

# 9. (اختیاری) برای پشتیبانی از فارسی و RTL در PyQt6
pip install qtawesome  # برای آیکون‌های زیبا و RTL
